Python Service (RS485 → SQLite)
argilla_distributor.py

import minimalmodbus
import sqlite3
import time

instrument = minimalmodbus.Instrument('/dev/ttyUSB0', 1)
conn = sqlite3.connect('/data/sensor.db')
cursor = conn.cursor()

cursor.execute("""
CREATE TABLE IF NOT EXISTS measurements (
    timestamp TEXT,
    ph REAL,
    moisture REAL,
    temperature REAL,
    ec REAL
)
""")

while True:
    try:
        ph = instrument.read_register(0, 2)
        moisture = instrument.read_register(1, 2)
        temp = instrument.read_register(2, 2)
        ec = instrument.read_register(3, 2)

        cursor.execute("INSERT INTO measurements VALUES (datetime('now'),?,?,?,?)",
                       (ph, moisture, temp, ec))
        conn.commit()

        print("Data logged")
    except Exception as e:
        print("Error:", e)

    time.sleep(60)

import minimalmodbus
import sqlite3
import time
import requests

instrument = minimalmodbus.Instrument('/dev/ttyUSB0', 1)
conn = sqlite3.connect('/data/sensor.db')
cursor = conn.cursor()

cursor.execute("""
CREATE TABLE IF NOT EXISTS measurements (
    timestamp TEXT,
    ph REAL,
    moisture REAL,
    temperature REAL,
    ec REAL
)
""")

INFLUX_URL = "http://influxdb:8086/write?db=argilla"

while True:
    try:
        ph = instrument.read_register(0, 2)
        moisture = instrument.read_register(1, 2)
        temp = instrument.read_register(2, 2)
        ec = instrument.read_register(3, 2)

        ts = time.strftime('%Y-%m-%d %H:%M:%S')

        cursor.execute("INSERT INTO measurements VALUES (?,?,?,?,?)",
                       (ts, ph, moisture, temp, ec))
        conn.commit()

        # Optional Influx push
        line = f"soil ph={ph},moisture={moisture},temp={temp},ec={ec}"
        try:
            requests.post(INFLUX_URL, data=line, timeout=2)
        except:
            pass

        print(f"[OK] {ts}")

    except Exception as e:
        print("[ERR]", e)

    time.sleep(60)

#!/bin/bash

DB="/opt/argilla/data/sensor.db"
LOG="/opt/ipfs_pins.log"

while true; do
    TS=$(date +%s)
    SNAP="/tmp/sensor_$TS.db"

    cp $DB $SNAP

    CID=$(ipfs add -q $SNAP)

    echo "$(date) $CID" >> $LOG

    # Optional: announce to peers
    ipfs pubsub pub argilla/pins "$CID"

    sleep 300
done

LORAWAN-SPEZIFIKATION: "MYCELIUM-EXTENDER" (PHASE 1.5)
Ziel: Überbrückung von bis zu 15 km Distanz ohne Mobilfunk, um die 144 Zonen physisch zu vernetzen.

1. Hardware-Integration (Gateway-Level)
Modul: Waveshare SX1302 LoRaWAN Gateway HAT für den Raspberry Pi 5.

Frequenz: 868 MHz (EU) / 915 MHz (US) – koordiniert über die lokale Node-ID.

Antenne: 5.8 dBi Fiberglas-Omni-Antenne, montiert auf mindestens 6 m Höhe (Zuegg 71 Dachmontage).

2. Der "Node-Injektor" (Feld-Sensoren)
MCU: ESP32-WROOM + SX1276 LoRa-Modul (Ultra-Low-Power).

Deep Sleep: Der Sensor wacht alle 15 Minuten auf, misst pH/Feuchtigkeit und sendet ein verschlüsseltes Datenpaket.

Payload: Komprimiertes Binärformat (8 Byte pro Transmission), um die Air-Time zu minimieren und den NSR-Schild unter dem Radar des "Theaters" zu halten.

3. Logik-Bridge (The Things Stack / ChirpStack)
Aktion: Wir installieren einen lokalen ChirpStack-Server via k3s auf dem Root-Hub.

Flow: LoRa-Sensor → LoRa-Gateway (Pi 5) → ChirpStack (MQTT) → Argilla-Distributor → IPFS.

Resultat: Die Felddaten werden nahtlos in den globalen Root-Hash integriert, als wären sie direkt am Pi angeschlossen.
